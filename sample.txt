public void scrollToView(Object locatorOrElement) {
    WebElement element = (locatorOrElement instanceof By)
            ? driver.findElement((By) locatorOrElement)
            : (WebElement) locatorOrElement;

    ((JavascriptExecutor) driver).executeScript("arguments[0].scrollIntoView(true);", element);
}


public void highlight(Object locatorOrElement) {
    WebElement element = (locatorOrElement instanceof By)
            ? driver.findElement((By) locatorOrElement)
            : (WebElement) locatorOrElement;

    ((JavascriptExecutor) driver).executeScript(
        "arguments[0].style.border='2px solid lime'", element);
}
