package com.cucumber.base;

import org.openqa.selenium.OutputType;
import org.openqa.selenium.TakesScreenshot;
import org.openqa.selenium.WebDriver;
import org.apache.commons.io.FileUtils;

import java.io.File;
import java.text.SimpleDateFormat;
import java.util.Date;

public class ScreenshotUtils {

    // ✅ Universal screenshot method (no need to pass driver everywhere)
    public static String takeScreenshot(String screenshotName) {
        String timeStamp = new SimpleDateFormat("yyMMdd_HHmmss").format(new Date());
        String destPath = "reports/screenshots/" + screenshotName + "_" + timeStamp + ".png";

        try {
            WebDriver driver = DriverFactory.getDriver(); // fetch driver from ThreadLocal
            if (driver != null) {
                File src = ((TakesScreenshot) driver).getScreenshotAs(OutputType.FILE);
                FileUtils.copyFile(src, new File(destPath));
            } else {
                System.out.println("⚠️ WebDriver instance is null, cannot take screenshot!");
            }
        } catch (Exception e) {
            e.printStackTrace();
        }

        return destPath;
    }

   
}
