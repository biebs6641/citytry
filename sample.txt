package com.mycompany.hooks;

import com.mycompany.base.ConfigReader;
import com.mycompany.base.DriverFactory;
import com.mycompany.base.ExtentManager;
import com.aventstack.extentreports.ExtentReports;
import com.aventstack.extentreports.ExtentTest;
import com.aventstack.extentreports.Status;
import com.aventstack.extentreports.MediaEntityBuilder;

import cucumber.api.Scenario;
import cucumber.api.java.After;
import cucumber.api.java.Before;

import org.openqa.selenium.OutputType;
import org.openqa.selenium.TakesScreenshot;

import java.io.File;
import java.io.IOException;
import java.nio.file.Files;

public class Hooks {
    private static ExtentReports extent;
    private static ThreadLocal<ExtentTest> test = new ThreadLocal<>();

    @Before
    public void setup(Scenario scenario) {
        extent = ExtentManager.getInstance();

        ExtentTest extentTest = extent.createTest(scenario.getName())
                .assignAuthor("Automation Team");

        // Assign tags as categories
        for (String tag : scenario.getSourceTagNames()) {
            extentTest.assignCategory(tag);
        }

        extentTest.log(Status.INFO, "Starting scenario: " + scenario.getName());
        test.set(extentTest);

        String browser = ConfigReader.getProperty("browser");
        DriverFactory.initDriver(browser);
    }

    @After
    public void teardown(Scenario scenario) {
        ExtentTest extentTest = test.get();

        if (scenario.isFailed()) {
            extentTest.log(Status.FAIL, "Scenario Failed: " + scenario.getName());

            try {
                File screenshot = ((TakesScreenshot) DriverFactory.getDriver())
                        .getScreenshotAs(OutputType.FILE);

                String screenshotPath = "test-output/screenshots/" + scenario.getName() + ".png";
                File dest = new File(screenshotPath);
                dest.getParentFile().mkdirs();
                Files.copy(screenshot.toPath(), dest.toPath());

                extentTest.fail("Screenshot of failure",
                        MediaEntityBuilder.createScreenCaptureFromPath(screenshotPath).build());

            } catch (IOException e) {
                extentTest.log(Status.WARNING, "Could not attach screenshot: " + e.getMessage());
            }
        } else {
            extentTest.log(Status.PASS, "Scenario Passed: " + scenario.getName());
        }

        DriverFactory.quitDriver();
        extent.flush();
    }
}
