public void waitForFullPageLoad(WebDriver driver) {
    WebDriverWait wait = new WebDriverWait(driver, Duration.ofSeconds(40));

    // Step 1: Wait for document ready state
    wait.until(driver1 -> ((JavascriptExecutor) driver1)
        .executeScript("return document.readyState").equals("complete"));

    // Step 2: Wait for no active XHR/Fetch requests
    wait.until(driver1 -> {
        JavascriptExecutor js = (JavascriptExecutor) driver1;
        Long pendingRequests = (Long) js.executeScript(
            "return window.__pendingRequests || 0"
        );
        return pendingRequests == 0;
    });
}
