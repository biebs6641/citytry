public void unhighlight(By locator) {
    try {
        WebElement element = wait.until(
                ExpectedConditions.presenceOfElementLocated(locator));

        jsExecutor.executeScript(
            "const el = arguments[0];" +

            // restore main element
            "if (el.hasAttribute('data-original-style')) {" +
            "   el.setAttribute('style', el.getAttribute('data-original-style'));" +
            "   el.removeAttribute('data-original-style');" +
            "}" +

            // restore children
            "const children = el.querySelectorAll('*');" +
            "children.forEach(c => {" +
            "   if (c.hasAttribute('data-original-style')) {" +
            "       c.setAttribute('style', c.getAttribute('data-original-style'));" +
            "       c.removeAttribute('data-original-style');" +
            "   }" +
            "});"
            , element);

    } catch (TimeoutException e) {
        System.out.println("Element not found for unhighlighting: " + locator);
    }
}
